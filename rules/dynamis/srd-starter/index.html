dyn:MovePreconditionAsk
  a dyn:SparqlAskPrecondition ;
  rdf:value """ASK {
  GRAPH ?stateGraph {
    VALUES (?actant ?fromPlace ?toPlace ?asOf) { (?actantValue ?fromPlaceValue ?toPlaceValue ?asOfValue) }
    ?fromPlace sc:adjacentTo ?toPlace .
    ?location a sc:LocatedInRelator ;
      sc:subject ?actant ;
      sc:predicate sc:locatedIn ;
      sc:object ?fromPlace ;
      sc:acceptedFrom ?acceptedFrom .
    FILTER(?acceptedFrom <= ?asOf)
    FILTER NOT EXISTS {
      ?location sc:acceptedUntil ?acceptedUntil .
      FILTER(?acceptedUntil <= ?asOf)
    }
  }
}""" .
